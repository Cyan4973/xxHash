688/******* Canonical representation *******/
689
690/*
691 * The default return values from XXH functions are unsigned 32 and 64 bit
692 * integers.
693 * This the simplest and fastest format for further post-processing.
694 *
695 * However, this leaves open the question of what is the order on the byte level,
696 * since little and big endian conventions will store the same number differently.
697 *
698 * The canonical representation settles this issue by mandating big-endian
699 * convention, the same convention as human-readable numbers (large digits first).
700 *
701 * When writing hash values to storage, sending them over a network, or printing
702 * them, it's highly recommended to use the canonical representation to ensure
703 * portability across a wider range of systems, present and future.
704 *
705 * The following functions allow transformation of hash values to and from
706 * canonical format.
707 */
708

713 unsigned char digest[4];
714} XXH32_canonical_t;

2568static xxh_u32 XXH32_round(xxh_u32 acc, xxh_u32 input)
2569{
2570 acc += input * XXH_PRIME32_2;
2571 acc = XXH_rotl32(acc, 13);
2572 acc *= XXH_PRIME32_1;
2573#if (defined(__SSE4_1__) || defined(__aarch64__) || defined(__wasm_simd128__)) && !defined(XXH_ENABLE_AUTOVECTORIZE)
2574 /*
2575 * UGLY HACK:
2576 * A compiler fence is the only thing that prevents GCC and Clang from
2577 * autovectorizing the XXH32 loop (pragmas and attributes don't work for some
2578 * reason) without globally disabling SSE4.1.
2579 *
2580 * The reason we want to avoid vectorization is because despite working on
2581 * 4 integers at a time, there are multiple factors slowing XXH32 down on
2582 * SSE4:
2583 * - There's a ridiculous amount of lag from pmulld (10 cycles of latency on
2584 * newer chips!) making it slightly slower to multiply four integers at
2585 * once compared to four integers independently. Even when pmulld was
2586 * fastest, Sandy/Ivy Bridge, it is still not worth it to go into SSE
2587 * just to multiply unless doing a long operation.
2588 *
2589 * - Four instructions are required to rotate,
2590 * movqda tmp, v // not required with VEX encoding
2591 * pslld tmp, 13 // tmp <<= 13
2592 * psrld v, 19 // x >>= 19
2593 * por v, tmp // x |= tmp
2594 * compared to one for scalar:
2595 * roll v, 13 // reliably fast across the board
2596 * shldl v, v, 13 // Sandy Bridge and later prefer this for some reason
2597 *
2598 * - Instruction level parallelism is actually more beneficial here because
2599 * the SIMD actually serializes this operation: While v1 is rotating, v2
2600 * can load data, while v3 can multiply. SSE forces them to operate
2601 * together.
2602 *
2603 * This is also enabled on AArch64, as Clang is *very aggressive* in vectorizing
2604 * the loop. NEON is only faster on the A53, and with the newer cores, it is less
2605 * than half the speed.
2606 *
2607 * Additionally, this is used on WASM SIMD128 because it JITs to the same
2608 * SIMD instructions and has the same issue.
2609 */
2610 XXH_COMPILER_GUARD(acc);
2611#endif
2612 return acc;
2613}

2625static xxh_u32 XXH32_avalanche(xxh_u32 hash)
2626{
2627 hash ^= hash >> 15;
2628 hash *= XXH_PRIME32_2;
2629 hash ^= hash >> 13;
2630 hash *= XXH_PRIME32_3;
2631 hash ^= hash >> 16;
2632 return hash;
2633}

2635#define XXH_get32bits(p) XXH_readLE32_align(p, align)

2652static XXH_PUREF xxh_u32
2653XXH32_finalize(xxh_u32 hash, const xxh_u8* ptr, size_t len, XXH_alignment align)
2654{
2655#define XXH_PROCESS1 do { \
2656 hash += (*ptr++) * XXH_PRIME32_5; \
2657 hash = XXH_rotl32(hash, 11) * XXH_PRIME32_1; \
2658} while (0)
2659
2660#define XXH_PROCESS4 do { \
2661 hash += XXH_get32bits(ptr) * XXH_PRIME32_3; \
2662 ptr += 4; \
2663 hash = XXH_rotl32(hash, 17) * XXH_PRIME32_4; \
2664} while (0)
2665
2666 if (ptr==NULL) XXH_ASSERT(len == 0);
2667
2668 /* Compact rerolled version; generally faster */
2669 if (!XXH32_ENDJMP) {
2670 len &= 15;
2671 while (len >= 4) {
2672 XXH_PROCESS4;
2673 len -= 4;
2674 }
2675 while (len > 0) {
2676 XXH_PROCESS1;
2677 --len;
2678 }
2679 return XXH32_avalanche(hash);
2680 } else {
2681 switch(len&15) /* or switch(bEnd - p) */ {
2682 case 12: XXH_PROCESS4;
2683 XXH_FALLTHROUGH; /* fallthrough */
2684 case 8: XXH_PROCESS4;
2685 XXH_FALLTHROUGH; /* fallthrough */
2686 case 4: XXH_PROCESS4;
2687 return XXH32_avalanche(hash);
2688
2689 case 13: XXH_PROCESS4;
2690 XXH_FALLTHROUGH; /* fallthrough */
2691 case 9: XXH_PROCESS4;
2692 XXH_FALLTHROUGH; /* fallthrough */
2693 case 5: XXH_PROCESS4;
2694 XXH_PROCESS1;
2695 return XXH32_avalanche(hash);
2696
2697 case 14: XXH_PROCESS4;
2698 XXH_FALLTHROUGH; /* fallthrough */
2699 case 10: XXH_PROCESS4;
2700 XXH_FALLTHROUGH; /* fallthrough */
2701 case 6: XXH_PROCESS4;
2702 XXH_PROCESS1;
2703 XXH_PROCESS1;
2704 return XXH32_avalanche(hash);
2705
2706 case 15: XXH_PROCESS4;
2707 XXH_FALLTHROUGH; /* fallthrough */
2708 case 11: XXH_PROCESS4;
2709 XXH_FALLTHROUGH; /* fallthrough */
2710 case 7: XXH_PROCESS4;
2711 XXH_FALLTHROUGH; /* fallthrough */
2712 case 3: XXH_PROCESS1;
2713 XXH_FALLTHROUGH; /* fallthrough */
2714 case 2: XXH_PROCESS1;
2715 XXH_FALLTHROUGH; /* fallthrough */
2716 case 1: XXH_PROCESS1;
2717 XXH_FALLTHROUGH; /* fallthrough */
2718 case 0: return XXH32_avalanche(hash);
2719 }
2720 XXH_ASSERT(0);
2721 return hash; /* reaching this point is deemed impossible */
2722 }
2723}

2741XXH_FORCE_INLINE XXH_PUREF xxh_u32
2742XXH32_endian_align(const xxh_u8* input, size_t len, xxh_u32 seed, XXH_alignment align)
2743{
2744 xxh_u32 h32;
2745
2746 if (input==NULL) XXH_ASSERT(len == 0);
2747
2748 if (len>=16) {
2749 const xxh_u8* const bEnd = input + len;
2750 const xxh_u8* const limit = bEnd - 15;
2751 xxh_u32 v1 = seed + XXH_PRIME32_1 + XXH_PRIME32_2;
2752 xxh_u32 v2 = seed + XXH_PRIME32_2;
2753 xxh_u32 v3 = seed + 0;
2754 xxh_u32 v4 = seed - XXH_PRIME32_1;
2755
2756 do {
2757 v1 = XXH32_round(v1, XXH_get32bits(input)); input += 4;
2758 v2 = XXH32_round(v2, XXH_get32bits(input)); input += 4;
2759 v3 = XXH32_round(v3, XXH_get32bits(input)); input += 4;
2760 v4 = XXH32_round(v4, XXH_get32bits(input)); input += 4;
2761 } while (input < limit);
2762
2763 h32 = XXH_rotl32(v1, 1) + XXH_rotl32(v2, 7)
2764 + XXH_rotl32(v3, 12) + XXH_rotl32(v4, 18);
2765 } else {
2766 h32 = seed + XXH_PRIME32_5;
2767 }
2768
2769 h32 += (xxh_u32)len;
2770
2771 return XXH32_finalize(h32, input, len&15, align);
2772}

2829XXH_PUBLIC_API XXH_errorcode
2830XXH32_update(XXH32_state_t* state, const void* input, size_t len)
2831{
2832 if (input==NULL) {
2833 XXH_ASSERT(len == 0);
2834 return XXH_OK;
2835 }
2836
2837 { const xxh_u8* p = (const xxh_u8*)input;
2838 const xxh_u8* const bEnd = p + len;
2839
2840 state->total_len_32 += (XXH32_hash_t)len;
2841 state->large_len |= (XXH32_hash_t)((len>=16) | (state->total_len_32>=16));
2842
2843 if (state->memsize + len < 16) { /* fill in tmp buffer */
2844 XXH_memcpy((xxh_u8*)(state->mem32) + state->memsize, input, len);
2845 state->memsize += (XXH32_hash_t)len;
2846 return XXH_OK;
2847 }
2848
2849 if (state->memsize) { /* some data left from previous update */
2850 XXH_memcpy((xxh_u8*)(state->mem32) + state->memsize, input, 16-state->memsize);
2851 { const xxh_u32* p32 = state->mem32;
2852 state->v[0] = XXH32_round(state->v[0], XXH_readLE32(p32)); p32++;
2853 state->v[1] = XXH32_round(state->v[1], XXH_readLE32(p32)); p32++;
2854 state->v[2] = XXH32_round(state->v[2], XXH_readLE32(p32)); p32++;
2855 state->v[3] = XXH32_round(state->v[3], XXH_readLE32(p32));
2856 }
2857 p += 16-state->memsize;
2858 state->memsize = 0;
2859 }
2860
2861 if (p <= bEnd-16) {
2862 const xxh_u8* const limit = bEnd - 16;
2863
2864 do {
2865 state->v[0] = XXH32_round(state->v[0], XXH_readLE32(p)); p+=4;
2866 state->v[1] = XXH32_round(state->v[1], XXH_readLE32(p)); p+=4;
2867 state->v[2] = XXH32_round(state->v[2], XXH_readLE32(p)); p+=4;
2868 state->v[3] = XXH32_round(state->v[3], XXH_readLE32(p)); p+=4;
2869 } while (p<=limit);
2870
2871 }
2872
2873 if (p < bEnd) {
2874 XXH_memcpy(state->mem32, p, (size_t)(bEnd-p));
2875 state->memsize = (unsigned)(bEnd-p);
2876 }
2877 }
2878
2879 return XXH_OK;
2880}

2885{

3430/*
3431 * One goal of XXH3 is to make it fast on both 32-bit and 64-bit, while
3432 * remaining a true 64-bit/128-bit hash function.
3433 *
3434 * This is done by prioritizing a subset of 64-bit operations that can be
3435 * emulated without too many steps on the average 32-bit machine.
3436 *
3437 * For example, these two lines seem similar, and run equally fast on 64-bit:
3438 *
3439 * xxh_u64 x;
3440 * x ^= (x >> 47); // good
3441 * x ^= (x >> 13); // bad
3442 *
3443 * However, to a 32-bit machine, there is a major difference.
3444 *
3445 * x ^= (x >> 47) looks like this:
3446 *
3447 * x.lo ^= (x.hi >> (47 - 32));
3448 *
3449 * while x ^= (x >> 13) looks like this:
3450 *
3451 * // note: funnel shifts are not usually cheap.
3452 * x.lo ^= (x.lo >> 13) | (x.hi << (32 - 13));
3453 * x.hi ^= (x.hi >> 13);
3454 *
3455 * The first one is significantly faster than the second, simply because the
3456 * shift is larger than 32. This means:
3457 * - All the bits we need are in the upper 32 bits, so we can ignore the lower
3458 * 32 bits in the shift.
3459 * - The shift result will always fit in the lower 32 bits, and therefore,
3460 * we can ignore the upper 32 bits in the xor.
3461 *
3462 * Thanks to this optimization, XXH3 only requires these features to be efficient:
3463 *
3464 * - Usable unaligned access
3465 * - A 32-bit or 64-bit ALU
3466 * - If 32-bit, a decent ADC instruction
3467 * - A 32 or 64-bit multiply with a 64-bit result
3468 * - For the 128-bit variant, a decent byteswap helps short inputs.
3469 *
3470 * The first two are already required by XXH32, and almost all 32-bit and 64-bit
3471 * platforms which can run XXH32 can run XXH3 efficiently.
3472 *
3473 * Thumb-1, the classic 16-bit only subset of ARM's instruction set, is one
3474 * notable exception.
3475 *
3476 * First of all, Thumb-1 lacks support for the UMULL instruction which
3477 * performs the important long multiply. This means numerous __aeabi_lmul
3478 * calls.
3479 *
3480 * Second of all, the 8 functional registers are just not enough.
3481 * Setup for __aeabi_lmul, byteshift loads, pointers, and all arithmetic need
3482 * Lo registers, and this shuffling results in thousands more MOVs than A32.
3483 *
3484 * A32 and T32 don't have this limitation. They can access all 14 registers,
3485 * do a 32->64 multiply with UMULL, and the flexible operand allowing free
3486 * shifts is helpful, too.
3487 *
3488 * Therefore, we do a quick sanity check.
3489 *
3490 * If compiling Thumb-1 for a target which supports ARM instructions, we will
3491 * emit a warning, as it is not a "sane" platform to compile for.
3492 *
3493 * Usually, if this happens, it is because of an accident and you probably need
3494 * to specify -march, as you likely meant to compile for a newer architecture.
3495 *
3496 * Credit: large sections of the vectorial and asm source code paths
3497 * have been contributed by @easyaspi314
3498 */

4024static XXH128_hash_t
4025XXH_mult64to128(xxh_u64 lhs, xxh_u64 rhs)
4026{
4027 /*
4028 * GCC/Clang __uint128_t method.
4029 *
4030 * On most 64-bit targets, GCC and Clang define a __uint128_t type.
4031 * This is usually the best way as it usually uses a native long 64-bit
4032 * multiply, such as MULQ on x86_64 or MUL + UMULH on aarch64.
4033 *
4034 * Usually.
4035 *
4036 * Despite being a 32-bit platform, Clang (and emscripten) define this type
4037 * despite not having the arithmetic for it. This results in a laggy
4038 * compiler builtin call which calculates a full 128-bit multiply.
4039 * In that case it is best to use the portable one.
4040 * https://github.com/Cyan4973/xxHash/issues/211#issuecomment-515575677
4041 */
4042#if (defined(__GNUC__) || defined(__clang__)) && !defined(__wasm__) \
4043 && defined(__SIZEOF_INT128__) \
4044 || (defined(_INTEGRAL_MAX_BITS) && _INTEGRAL_MAX_BITS >= 128)
4045
4046 __uint128_t const product = (__uint128_t)lhs * (__uint128_t)rhs;
4047 XXH128_hash_t r128;
4048 r128.low64 = (xxh_u64)(product);
4049 r128.high64 = (xxh_u64)(product >> 64);
4050 return r128;
4051
4052 /*
4053 * MSVC for x64's _umul128 method.
4054 *
4055 * xxh_u64 _umul128(xxh_u64 Multiplier, xxh_u64 Multiplicand, xxh_u64 *HighProduct);
4056 *
4057 * This compiles to single operand MUL on x64.
4058 */
4059#elif (defined(_M_X64) || defined(_M_IA64)) && !defined(_M_ARM64EC)
4060
4061#ifndef _MSC_VER
4062# pragma intrinsic(_umul128)
4063#endif
4064 xxh_u64 product_high;
4065 xxh_u64 const product_low = _umul128(lhs, rhs, &product_high);
4066 XXH128_hash_t r128;
4067 r128.low64 = product_low;
4068 r128.high64 = product_high;
4069 return r128;
4070
4071 /*
4072 * MSVC for ARM64's __umulh method.
4073 *
4074 * This compiles to the same MUL + UMULH as GCC/Clang's __uint128_t method.
4075 */
4076#elif defined(_M_ARM64) || defined(_M_ARM64EC)
4077
4078#ifndef _MSC_VER
4079# pragma intrinsic(__umulh)
4080#endif
4081 XXH128_hash_t r128;
4082 r128.low64 = lhs * rhs;
4083 r128.high64 = __umulh(lhs, rhs);
4084 return r128;
4085
4086#else
4087 /*
4088 * Portable scalar method. Optimized for 32-bit and 64-bit ALUs.
4089 *
4090 * This is a fast and simple grade school multiply, which is shown below
4091 * with base 10 arithmetic instead of base 0x100000000.
4092 *
4093 * 9 3 // D2 lhs = 93
4094 * x 7 5 // D2 rhs = 75
4095 * ----------
4096 * 1 5 // D2 lo_lo = (93 % 10) * (75 % 10) = 15
4097 * 4 5 | // D2 hi_lo = (93 / 10) * (75 % 10) = 45
4098 * 2 1 | // D2 lo_hi = (93 % 10) * (75 / 10) = 21
4099 * + 6 3 | | // D2 hi_hi = (93 / 10) * (75 / 10) = 63
4100 * ---------
4101 * 2 7 | // D2 cross = (15 / 10) + (45 % 10) + 21 = 27
4102 * + 6 7 | | // D2 upper = (27 / 10) + (45 / 10) + 63 = 67
4103 * ---------
4104 * 6 9 7 5 // D4 res = (27 * 10) + (15 % 10) + (67 * 100) = 6975
4105 *
4106 * The reasons for adding the products like this are:
4107 * 1. It avoids manual carry tracking. Just like how
4108 * (9 * 9) + 9 + 9 = 99, the same applies with this for UINT64_MAX.
4109 * This avoids a lot of complexity.
4110 *
4111 * 2. It hints for, and on Clang, compiles to, the powerful UMAAL
4112 * instruction available in ARM's Digital Signal Processing extension
4113 * in 32-bit ARMv6 and later, which is shown below:
4114 *
4115 * void UMAAL(xxh_u32 *RdLo, xxh_u32 *RdHi, xxh_u32 Rn, xxh_u32 Rm)
4116 * {
4117 * xxh_u64 product = (xxh_u64)*RdLo * (xxh_u64)*RdHi + Rn + Rm;
4118 * *RdLo = (xxh_u32)(product & 0xFFFFFFFF);
4119 * *RdHi = (xxh_u32)(product >> 32);
4120 * }
4121 *
4122 * This instruction was designed for efficient long multiplication, and
4123 * allows this to be calculated in only 4 instructions at speeds
4124 * comparable to some 64-bit ALUs.
4125 *
4126 * 3. It isn't terrible on other platforms. Usually this will be a couple
4127 * of 32-bit ADD/ADCs.
4128 */
4129
4130 /* First calculate all of the cross products. */
4131 xxh_u64 const lo_lo = XXH_mult32to64(lhs & 0xFFFFFFFF, rhs & 0xFFFFFFFF);
4132 xxh_u64 const hi_lo = XXH_mult32to64(lhs >> 32, rhs & 0xFFFFFFFF);
4133 xxh_u64 const lo_hi = XXH_mult32to64(lhs & 0xFFFFFFFF, rhs >> 32);
4134 xxh_u64 const hi_hi = XXH_mult32to64(lhs >> 32, rhs >> 32);
4135
4136 /* Now add the products together. These will never overflow. */
4137 xxh_u64 const cross = (lo_lo >> 32) + (hi_lo & 0xFFFFFFFF) + lo_hi;
4138 xxh_u64 const upper = (hi_lo >> 32) + (cross >> 32) + hi_hi;
4139 xxh_u64 const lower = (cross << 32) | (lo_lo & 0xFFFFFFFF);
4140
4141 XXH128_hash_t r128;
4142 r128.low64 = lower;
4143 r128.high64 = upper;
4144 return r128;
4145#endif
4146}

4200/* ==========================================
4201 * Short keys
4202 * ==========================================
4203 * One of the shortcomings of XXH32 and XXH64 was that their performance was
4204 * sub-optimal on short lengths. It used an iterative algorithm which strongly
4205 * favored lengths that were a multiple of 4 or 8.
4206 *
4207 * Instead of iterating over individual inputs, we use a set of single shot
4208 * functions which piece together a range of lengths and operate in constant time.
4209 *
4210 * Additionally, the number of multiplies has been significantly reduced. This
4211 * reduces latency, especially when emulating 64-bit multiplies on 32-bit.
4212 *
4213 * Depending on the platform, this may or may not be faster than XXH32, but it
4214 * is almost guaranteed to be faster than XXH64.
4215 */
4216
4217/*
4218 * At very short lengths, there isn't enough input to fully hide secrets, or use
4219 * the entire secret.
4220 *
4221 * There is also only a limited amount of mixing we can do before significantly
4222 * impacting performance.
4223 *
4224 * Therefore, we use different sections of the secret and always mix two secret
4225 * samples with an XOR. This should have no effect on performance on the
4226 * seedless or withSeed variants because everything _should_ be constant folded
4227 * by modern compilers.
4228 *
4229 * The XOR mixing hides individual parts of the secret and increases entropy.
4230 *
4231 * This adds an extra layer of strength for custom secrets.
4232 */
4233XXH_FORCE_INLINE XXH_PUREF XXH64_hash_t
4234XXH3_len_1to3_64b(const xxh_u8* input, size_t len, const xxh_u8* secret, XXH64_hash_t seed)
4235{
4236 XXH_ASSERT(input != NULL);
4237 XXH_ASSERT(1 <= len && len <= 3);
4238 XXH_ASSERT(secret != NULL);
4239 /*
4240 * len = 1: combined = { input[0], 0x01, input[0], input[0] }
4241 * len = 2: combined = { input[1], 0x02, input[0], input[1] }
4242 * len = 3: combined = { input[2], 0x03, input[0], input[1] }
4243 */
4244 { xxh_u8 const c1 = input[0];
4245 xxh_u8 const c2 = input[len >> 1];
4246 xxh_u8 const c3 = input[len - 1];
4247 xxh_u32 const combined = ((xxh_u32)c1 << 16) | ((xxh_u32)c2 << 24)
4248 | ((xxh_u32)c3 << 0) | ((xxh_u32)len << 8);
4249 xxh_u64 const bitflip = (XXH_readLE32(secret) ^ XXH_readLE32(secret+4)) + seed;
4250 xxh_u64 const keyed = (xxh_u64)combined ^ bitflip;
4251 return XXH64_avalanche(keyed);
4252 }
4253}

4260 XXH_ASSERT(4 <= len && len <= 8);

4301 * multiplication by zero, affecting hashes of lengths 17 to 240.
4302 *
4303 * However, they are very unlikely.
4304 *
4305 * Keep this in mind when using the unseeded XXH3_64bits() variant: As with all
4306 * unseeded non-cryptographic hashes, it does not attempt to defend itself
4307 * against specially crafted inputs, only random inputs.
4308 *
4309 * Compared to classic UMAC where a 1 in 2^31 chance of 4 consecutive bytes
4310 * cancelling out the secret is taken an arbitrary number of times (addressed
4311 * in XXH3_accumulate_512), this collision is very unlikely with random inputs
4312 * and/or proper seeding:
4313 *
4314 * This only has a 1 in 2^63 chance of 8 consecutive bytes cancelling out, in a
4315 * function that is only called up to 16 times per hash with up to 240 bytes of
4316 * input.
4317 *
4318 * This is not too bad for a non-cryptographic hash function, especially with
4319 * only 64 bit outputs.
4320 *
4321 * The 128-bit variant (which trades some speed for strength) is NOT affected
4322 * by this, although it is always a good idea to use a proper seed if you care
4323 * about strength.
4324 */

4326 const xxh_u8* XXH_RESTRICT secret, xxh_u64 seed64)
4327{
4328#if defined(__GNUC__) && !defined(__clang__) /* GCC, not Clang */ \

4330 && !defined(XXH_ENABLE_AUTOVECTORIZE) /* Define to disable like XXH32 hack */
4331 /*

4335 *
4336 * By forcing seed64 into a register, we disrupt the cost model and
4337 * cause it to scalarize. See `XXH32_round()`
4338 *
4339 * FIXME: Clang's output is still _much_ faster -- On an AMD Ryzen 3600,
4340 * XXH3_64bits @ len=240 runs at 4.6 GB/s with Clang 9, but 3.3 GB/s on

4342 *
4343 * GCC generates much better scalar code than Clang for the rest of XXH3,
4344 * which is why finding a more optimal codepath is an interest.

4531/*
4532 * XXH3_accumulate_512 is the tightest loop for long inputs, and it is the most optimized.
4533 *
4534 * It is a hardened version of UMAC, based off of FARSH's implementation.
4535 *
4536 * This was chosen because it adapts quite well to 32-bit, 64-bit, and SIMD
4537 * implementations, and it is ridiculously fast.
4538 *
4539 * We harden it by mixing the original input to the accumulators as well as the product.
4540 *
4541 * This means that in the (relatively likely) case of a multiply by zero, the
4542 * original input is preserved.
4543 *
4544 * On 128-bit inputs, we swap 64-bit pairs when we add the input to improve
4545 * cross-pollination, as otherwise the upper and lower halves would be
4546 * essentially independent.
4547 *
4548 * This doesn't matter on 64-bit hashes since they all get merged together in
4549 * the end, so we skip the extra step.
4550 *
4551 * Both XXH3_64bits and XXH3_128bits use this subroutine.
4552 */

4561XXH_FORCE_INLINE XXH_TARGET_AVX512 void

4563 const void* XXH_RESTRICT input,
4564 const void* XXH_RESTRICT secret)
4565{

4567 XXH_ASSERT((((size_t)acc) & 63) == 0);
4568 XXH_STATIC_ASSERT(XXH_STRIPE_LEN == sizeof(__m512i));
4569
4570 {
4571 /* data_vec = input[0]; */
4572 __m512i const data_vec = _mm512_loadu_si512 (input);
4573 /* key_vec = secret[0]; */
4574 __m512i const key_vec = _mm512_loadu_si512 (secret);
4575 /* data_key = data_vec ^ key_vec; */
4576 __m512i const data_key = _mm512_xor_si512 (data_vec, key_vec);
4577 /* data_key_lo = data_key >> 32; */
4578 __m512i const data_key_lo = _mm512_srli_epi64 (data_key, 32);
4579 /* product = (data_key & 0xffffffff) * (data_key_lo & 0xffffffff); */
4580 __m512i const product = _mm512_mul_epu32 (data_key, data_key_lo);
4581 /* xacc[0] += swap(data_vec); */
4582 __m512i const data_swap = _mm512_shuffle_epi32(data_vec, (_MM_PERM_ENUM)_MM_SHUFFLE(1, 0, 3, 2));
4583 __m512i const sum = _mm512_add_epi64(*xacc, data_swap);
4584 /* xacc[0] += product; */

4586 }

4592 *
4593 * Multiplication isn't perfect, as explained by Google in HighwayHash:
4594 *
4595 * // Multiplication mixes/scrambles bytes 0-7 of the 64-bit result to
4596 * // varying degrees. In descending order of goodness, bytes
4597 * // 3 4 2 5 1 6 0 7 have quality 228 224 164 160 100 96 36 32.
4598 * // As expected, the upper and lower bytes are much worse.
4599 *
4600 * Source: https://github.com/google/highwayhash/blob/0aaf66b/highwayhash/hh_avx2.h#L291
4601 *
4602 * Since our algorithm uses a pseudorandom secret to add some variance into the
4603 * mix, we don't need to (or want to) mix as often or as much as HighwayHash does.
4604 *
4605 * This isn't as tight as XXH3_accumulate, but still written in SIMD to avoid
4606 * extraction.
4607 *
4608 * Both XXH3_64bits and XXH3_128bits use this subroutine.
4609 */

4613{

4615 XXH_STATIC_ASSERT(XXH_STRIPE_LEN == sizeof(__m512i));
4616 { __m512i* const xacc = (__m512i*) acc;
4617 const __m512i prime32 = _mm512_set1_epi32((int)XXH_PRIME32_1);
4618
4619 /* xacc[0] ^= (xacc[0] >> 47) */
4620 __m512i const acc_vec = *xacc;
4621 __m512i const shifted = _mm512_srli_epi64 (acc_vec, 47);
4622 /* xacc[0] ^= secret; */
4623 __m512i const key_vec = _mm512_loadu_si512 (secret);
4624 __m512i const data_key = _mm512_ternarylogic_epi32(key_vec, acc_vec, shifted, 0x96 /* key_vec ^ acc_vec ^ shifted */);

4626 /* xacc[0] *= XXH_PRIME32_1; */
4627 __m512i const data_key_hi = _mm512_srli_epi64 (data_key, 32);
4628 __m512i const prod_lo = _mm512_mul_epu32 (data_key, prime32);
4629 __m512i const prod_hi = _mm512_mul_epu32 (data_key_hi, prime32);
4630 *xacc = _mm512_add_epi64(prod_lo, _mm512_slli_epi64(prod_hi, 32));

4906XXH3_accumulate_512_neon( void* XXH_RESTRICT acc,
4907 const void* XXH_RESTRICT input,
4908 const void* XXH_RESTRICT secret)
4909{
4910 XXH_ASSERT((((size_t)acc) & 15) == 0);
4911 XXH_STATIC_ASSERT(XXH3_NEON_LANES > 0 && XXH3_NEON_LANES <= XXH_ACC_NB && XXH3_NEON_LANES % 2 == 0);
4912 { /* GCC for darwin arm64 does not like aliasing here */
4913 xxh_aliasing_uint64x2_t* const xacc = (xxh_aliasing_uint64x2_t*) acc;
4914 /* We don't use a uint32x4_t pointer because it causes bus errors on ARMv7. */
4915 uint8_t const* xinput = (const uint8_t *) input;
4916 uint8_t const* xsecret = (const uint8_t *) secret;
4917
4918 size_t i;
4919#ifdef __wasm_simd128__
4920 /*
4921 * On WASM SIMD128, Clang emits direct address loads when XXH3_kSecret
4922 * is constant propagated, which results in it converting it to this
4923 * inside the loop:
4924 *
4925 * a = v128.load(XXH3_kSecret + 0 + $secret_offset, offset = 0)
4926 * b = v128.load(XXH3_kSecret + 16 + $secret_offset, offset = 0)
4927 * ...
4928 *
4929 * This requires a full 32-bit address immediate (and therefore a 6 byte
4930 * instruction) as well as an add for each offset.
4931 *
4932 * Putting an asm guard prevents it from folding (at the cost of losing
4933 * the alignment hint), and uses the free offset in `v128.load` instead
4934 * of adding secret_offset each time which overall reduces code size by
4935 * about a kilobyte and improves performance.
4936 */
4937 XXH_COMPILER_GUARD(xsecret);
4938#endif
4939 /* Scalar lanes use the normal scalarRound routine */
4940 for (i = XXH3_NEON_LANES; i < XXH_ACC_NB; i++) {
4941 XXH3_scalarRound(acc, input, secret, i);
4942 }
4943 i = 0;
4944 /* 4 NEON lanes at a time. */
4945 for (; i+1 < XXH3_NEON_LANES / 2; i+=2) {
4946 /* data_vec = xinput[i]; */
4947 uint64x2_t data_vec_1 = XXH_vld1q_u64(xinput + (i * 16));
4948 uint64x2_t data_vec_2 = XXH_vld1q_u64(xinput + ((i+1) * 16));
4949 /* key_vec = xsecret[i]; */
4950 uint64x2_t key_vec_1 = XXH_vld1q_u64(xsecret + (i * 16));
4951 uint64x2_t key_vec_2 = XXH_vld1q_u64(xsecret + ((i+1) * 16));
4952 /* data_swap = swap(data_vec) */
4953 uint64x2_t data_swap_1 = vextq_u64(data_vec_1, data_vec_1, 1);
4954 uint64x2_t data_swap_2 = vextq_u64(data_vec_2, data_vec_2, 1);
4955 /* data_key = data_vec ^ key_vec; */
4956 uint64x2_t data_key_1 = veorq_u64(data_vec_1, key_vec_1);
4957 uint64x2_t data_key_2 = veorq_u64(data_vec_2, key_vec_2);
4958
4959 /*
4960 * If we reinterpret the 64x2 vectors as 32x4 vectors, we can use a
4961 * de-interleave operation for 4 lanes in 1 step with `vuzpq_u32` to
4962 * get one vector with the low 32 bits of each lane, and one vector
4963 * with the high 32 bits of each lane.
4964 *
4965 * The intrinsic returns a double vector because the original ARMv7-a
4966 * instruction modified both arguments in place. AArch64 and SIMD128 emit
4967 * two instructions from this intrinsic.
4968 *
4969 * [ dk11L | dk11H | dk12L | dk12H ] -> [ dk11L | dk12L | dk21L | dk22L ]
4970 * [ dk21L | dk21H | dk22L | dk22H ] -> [ dk11H | dk12H | dk21H | dk22H ]
4971 */
4972 uint32x4x2_t unzipped = vuzpq_u32(
4973 vreinterpretq_u32_u64(data_key_1),
4974 vreinterpretq_u32_u64(data_key_2)
4975 );
4976 /* data_key_lo = data_key & 0xFFFFFFFF */
4977 uint32x4_t data_key_lo = unzipped.val[0];
4978 /* data_key_hi = data_key >> 32 */
4979 uint32x4_t data_key_hi = unzipped.val[1];
4980 /*
4981 * Then, we can split the vectors horizontally and multiply which, as for most
4982 * widening intrinsics, have a variant that works on both high half vectors
4983 * for free on AArch64. A similar instruction is available on SIMD128.
4984 *
4985 * sum = data_swap + (u64x2) data_key_lo * (u64x2) data_key_hi
4986 */
4987 uint64x2_t sum_1 = XXH_vmlal_low_u32(data_swap_1, data_key_lo, data_key_hi);
4988 uint64x2_t sum_2 = XXH_vmlal_high_u32(data_swap_2, data_key_lo, data_key_hi);
4989 /*
4990 * Clang reorders
4991 * a += b * c; // umlal swap.2d, dkl.2s, dkh.2s
4992 * c += a; // add acc.2d, acc.2d, swap.2d
4993 * to
4994 * c += a; // add acc.2d, acc.2d, swap.2d
4995 * c += b * c; // umlal acc.2d, dkl.2s, dkh.2s
4996 *
4997 * While it would make sense in theory since the addition is faster,
4998 * for reasons likely related to umlal being limited to certain NEON
4999 * pipelines, this is worse. A compiler guard fixes this.
5000 */
5001 XXH_COMPILER_GUARD_CLANG_NEON(sum_1);
5002 XXH_COMPILER_GUARD_CLANG_NEON(sum_2);
5003 /* xacc[i] = acc_vec + sum; */
5004 xacc[i] = vaddq_u64(xacc[i], sum_1);
5005 xacc[i+1] = vaddq_u64(xacc[i+1], sum_2);
5006 }
5007 /* Operate on the remaining NEON lanes 2 at a time. */
5008 for (; i < XXH3_NEON_LANES / 2; i++) {
5009 /* data_vec = xinput[i]; */
5010 uint64x2_t data_vec = XXH_vld1q_u64(xinput + (i * 16));
5011 /* key_vec = xsecret[i]; */
5012 uint64x2_t key_vec = XXH_vld1q_u64(xsecret + (i * 16));
5013 /* acc_vec_2 = swap(data_vec) */
5014 uint64x2_t data_swap = vextq_u64(data_vec, data_vec, 1);
5015 /* data_key = data_vec ^ key_vec; */
5016 uint64x2_t data_key = veorq_u64(data_vec, key_vec);
5017 /* For two lanes, just use VMOVN and VSHRN. */
5018 /* data_key_lo = data_key & 0xFFFFFFFF; */
5019 uint32x2_t data_key_lo = vmovn_u64(data_key);
5020 /* data_key_hi = data_key >> 32; */
5021 uint32x2_t data_key_hi = vshrn_n_u64(data_key, 32);
5022 /* sum = data_swap + (u64x2) data_key_lo * (u64x2) data_key_hi; */
5023 uint64x2_t sum = vmlal_u32(data_swap, data_key_lo, data_key_hi);
5024 /* Same Clang workaround as before */
5025 XXH_COMPILER_GUARD_CLANG_NEON(sum);
5026 /* xacc[i] = acc_vec + sum; */
5027 xacc[i] = vaddq_u64 (xacc[i], sum);
5028 }
5029 }
5030}

5037

5039 uint8_t const* xsecret = (uint8_t const*) secret;
5040

5045 uint32x2_t const kPrimeLo = vdup_n_u32(XXH_PRIME32_1);
5046 /* { 0, prime32_1, 0, prime32_1 } */
5047 uint32x4_t const kPrimeHi = vreinterpretq_u32_u64(vdupq_n_u64((xxh_u64)XXH_PRIME32_1 << 32));
5048#endif

5058 uint64x2_t data_vec = veorq_u64(acc_vec, shifted);

5067#else

5069 * Expanded version with portable NEON intrinsics
5070 *
5071 * lo(x) * lo(y) + (hi(x) * lo(y) << 32)
5072 *
5073 * prod_hi = hi(data_key) * lo(prime) << 32
5074 *
5075 * Since we only need 32 bits of this multiply a trick can be used, reinterpreting the vector
5076 * as a uint32x4_t and multiplying by { 0, prime, 0, prime } to cancel out the unwanted bits
5077 * and avoid the shift.
5078 */

5080 /* Extract low bits for vmlal_u32 */
5081 uint32x2_t data_key_lo = vmovn_u64(data_key);
5082 /* xacc[i] = prod_hi + lo(data_key) * XXH_PRIME32_1; */

5390XXH_FORCE_INLINE void
5391XXH3_initCustomSecret_scalar(void* XXH_RESTRICT customSecret, xxh_u64 seed64)
5392{
5393 /*
5394 * We need a separate pointer for the hack below,
5395 * which requires a non-const pointer.
5396 * Any decent compiler will optimize this out otherwise.
5397 */
5398 const xxh_u8* kSecretPtr = XXH3_kSecret;
5399 XXH_STATIC_ASSERT((XXH_SECRET_DEFAULT_SIZE & 15) == 0);
5400
5401#if defined(__GNUC__) && defined(__aarch64__)
5402 /*
5403 * UGLY HACK:
5404 * GCC and Clang generate a bunch of MOV/MOVK pairs for aarch64, and they are
5405 * placed sequentially, in order, at the top of the unrolled loop.
5406 *
5407 * While MOVK is great for generating constants (2 cycles for a 64-bit
5408 * constant compared to 4 cycles for LDR), it fights for bandwidth with
5409 * the arithmetic instructions.
5410 *
5411 * I L S
5412 * MOVK
5413 * MOVK
5414 * MOVK
5415 * MOVK
5416 * ADD
5417 * SUB STR
5418 * STR
5419 * By forcing loads from memory (as the asm line causes the compiler to assume
5420 * that XXH3_kSecretPtr has been changed), the pipelines are used more
5421 * efficiently:
5422 * I L S
5423 * LDR
5424 * ADD LDR
5425 * SUB STR
5426 * STR
5427 *
5428 * See XXH3_NEON_LANES for details on the pipsline.
5429 *
5430 * XXH3_64bits_withSeed, len == 256, Snapdragon 835
5431 * without hack: 2654.4 MB/s
5432 * with hack: 3202.9 MB/s
5433 */
5434 XXH_COMPILER_GUARD(kSecretPtr);
5435#endif
5436 { int const nbRounds = XXH_SECRET_DEFAULT_SIZE / 16;
5437 int i;
5438 for (i=0; i < nbRounds; i++) {
5439 /*
5440 * The asm hack causes the compiler to assume that kSecretPtr aliases with
5441 * customSecret, and on aarch64, this prevented LDP from merging two
5442 * loads together for free. Putting the loads together before the stores
5443 * properly generates LDP.
5444 */
5445 xxh_u64 lo = XXH_readLE64(kSecretPtr + 16*i) + seed64;
5446 xxh_u64 hi = XXH_readLE64(kSecretPtr + 16*i + 8) - seed64;
5447 XXH_writeLE64((xxh_u8*)customSecret + 16*i, lo);
5448 XXH_writeLE64((xxh_u8*)customSecret + 16*i + 8, hi);
5449 } }
5450}

5513XXH_FORCE_INLINE void
5514XXH3_hashLong_internal_loop(xxh_u64* XXH_RESTRICT acc,
5515 const xxh_u8* XXH_RESTRICT input, size_t len,
5516 const xxh_u8* XXH_RESTRICT secret, size_t secretSize,
5517 XXH3_f_accumulate f_acc,
5518 XXH3_f_scrambleAcc f_scramble)
5519{
5520 size_t const nbStripesPerBlock = (secretSize - XXH_STRIPE_LEN) / XXH_SECRET_CONSUME_RATE;
5521 size_t const block_len = XXH_STRIPE_LEN * nbStripesPerBlock;
5522 size_t const nb_blocks = (len - 1) / block_len;
5523
5524 size_t n;
5525
5526 XXH_ASSERT(secretSize >= XXH3_SECRET_SIZE_MIN);
5527
5528 for (n = 0; n < nb_blocks; n++) {
5529 f_acc(acc, input + n*block_len, secret, nbStripesPerBlock);
5530 f_scramble(acc, secret + secretSize - XXH_STRIPE_LEN);
5531 }
5532
5533 /* last partial block */
5534 XXH_ASSERT(len > XXH_STRIPE_LEN);
5535 { size_t const nbStripes = ((len - 1) - (block_len * nb_blocks)) / XXH_STRIPE_LEN;
5536 XXH_ASSERT(nbStripes <= (secretSize / XXH_SECRET_CONSUME_RATE));
5537 f_acc(acc, input + nb_blocks*block_len, secret, nbStripes);
5538
5539 /* last stripe */
5540 { const xxh_u8* const p = input + len - XXH_STRIPE_LEN;
5541#define XXH_SECRET_LASTACC_START 7 /* not aligned on 8, last secret is different from acc & scrambler */
5542 XXH3_accumulate_512(acc, p, secret + secretSize - XXH_STRIPE_LEN - XXH_SECRET_LASTACC_START);
5543 } }

5734/*
5735 * Malloc's a pointer that is always aligned to align.
5736 *
5737 * This must be freed with `XXH_alignedFree()`.
5738 *
5739 * malloc typically guarantees 16 byte alignment on 64-bit systems and 8 byte
5740 * alignment on 32-bit. This isn't enough for the 32 byte aligned loads in AVX2
5741 * or on 32-bit, the 16 byte aligned loads in SSE2 and NEON.
5742 *
5743 * This underalignment previously caused a rather obvious crash which went
5744 * completely unnoticed due to XXH3_createState() not actually being tested.
5745 * Credit to RedSpah for noticing this bug.
5746 *
5747 * The alignment is done manually: Functions like posix_memalign or _mm_malloc
5748 * are avoided: To maintain portability, we would have to write a fallback
5749 * like this anyways, and besides, testing for the existence of library
5750 * functions without relying on external build tools is impossible.
5751 *
5752 * The method is simple: Overallocate, manually align, and store the offset
5753 * to the original behind the returned pointer.
5754 *
5755 * Align must be a power of 2 and 8 <= align <= 128.
5756 */
5757static XXH_MALLOCF void* XXH_alignedMalloc(size_t s, size_t align)
5758{
5759 XXH_ASSERT(align <= 128 && align >= 8); /* range check */
5760 XXH_ASSERT((align & (align-1)) == 0); /* power of 2 */
5761 XXH_ASSERT(s != 0 && s < (s + align)); /* empty/overflow */
5762 { /* Overallocate to make room for manual realignment and an offset byte */
5763 xxh_u8* base = (xxh_u8*)XXH_malloc(s + align);
5764 if (base != NULL) {
5765 /*
5766 * Get the offset needed to align this pointer.
5767 *
5768 * Even if the returned pointer is aligned, there will always be
5769 * at least one byte to store the offset to the original pointer.
5770 */
5771 size_t offset = align - ((size_t)base & (align - 1)); /* base % align */
5772 /* Add the offset for the now-aligned pointer */
5773 xxh_u8* ptr = base + offset;
5774
5775 XXH_ASSERT((size_t)ptr % align == 0);
5776
5777 /* Store the offset immediately before the returned pointer. */
5778 ptr[-1] = (xxh_u8)offset;
5779 return ptr;
5780 }
5781 return NULL;
5782 }
5783}
5784/*
5785 * Frees an aligned pointer allocated by XXH_alignedMalloc(). Don't pass
5786 * normal malloc'd pointers, XXH_alignedMalloc has a specific data layout.
5787 */
5788static void XXH_alignedFree(void* p)
5789{
5790 if (p != NULL) {
5791 xxh_u8* ptr = (xxh_u8*)p;
5792 /* Get the offset byte we added in XXH_malloc. */
5793 xxh_u8 offset = ptr[-1];
5794 /* Free the original malloc'd pointer */
5795 xxh_u8* base = ptr - offset;
5796 XXH_free(base);
5797 }

5923XXH_FORCE_INLINE const xxh_u8 *
5924XXH3_consumeStripes(xxh_u64* XXH_RESTRICT acc,
5925 size_t* XXH_RESTRICT nbStripesSoFarPtr, size_t nbStripesPerBlock,
5926 const xxh_u8* XXH_RESTRICT input, size_t nbStripes,
5927 const xxh_u8* XXH_RESTRICT secret, size_t secretLimit,
5928 XXH3_f_accumulate f_acc,
5929 XXH3_f_scrambleAcc f_scramble)
5930{
5931 const xxh_u8* initialSecret = secret + *nbStripesSoFarPtr * XXH_SECRET_CONSUME_RATE;
5932 /* Process full blocks */
5933 if (nbStripes >= (nbStripesPerBlock - *nbStripesSoFarPtr)) {
5934 /* Process the initial partial block... */
5935 size_t nbStripesThisIter = nbStripesPerBlock - *nbStripesSoFarPtr;
5936
5937 do {
5938 /* Accumulate and scramble */
5939 f_acc(acc, input, initialSecret, nbStripesThisIter);
5940 f_scramble(acc, secret + secretLimit);
5941 input += nbStripesThisIter * XXH_STRIPE_LEN;
5942 nbStripes -= nbStripesThisIter;
5943 /* Then continue the loop with the full block size */
5944 nbStripesThisIter = nbStripesPerBlock;
5945 initialSecret = secret;
5946 } while (nbStripes >= nbStripesPerBlock);
5947 *nbStripesSoFarPtr = 0;
5948 }
5949 /* Process a partial block */
5950 if (nbStripes > 0) {
5951 f_acc(acc, input, initialSecret, nbStripes);
5952 input += nbStripes * XXH_STRIPE_LEN;
5953 *nbStripesSoFarPtr += nbStripes;
5954 }
5955 /* Return end pointer */
5956 return input;
5957}

5964/*
5965 * Both XXH3_64bits_update and XXH3_128bits_update use this routine.
5966 */
5967XXH_FORCE_INLINE XXH_errorcode
5968XXH3_update(XXH3_state_t* XXH_RESTRICT const state,
5969 const xxh_u8* XXH_RESTRICT input, size_t len,
5970 XXH3_f_accumulate f_acc,
5971 XXH3_f_scrambleAcc f_scramble)
5972{
5973 if (input==NULL) {
5974 XXH_ASSERT(len == 0);
5975 return XXH_OK;
5976 }
5977
5978 XXH_ASSERT(state != NULL);
5979 { const xxh_u8* const bEnd = input + len;
5980 const unsigned char* const secret = (state->extSecret == NULL) ? state->customSecret : state->extSecret;
5981#if defined(XXH3_STREAM_USE_STACK) && XXH3_STREAM_USE_STACK >= 1
5982 /* For some reason, gcc and MSVC seem to suffer greatly
5983 * when operating accumulators directly into state.
5984 * Operating into stack space seems to enable proper optimization.
5985 * clang, on the other hand, doesn't seem to need this trick */
5986 XXH_ALIGN(XXH_ACC_ALIGN) xxh_u64 acc[8];
5987 XXH_memcpy(acc, state->acc, sizeof(acc));
5988#else
5989 xxh_u64* XXH_RESTRICT const acc = state->acc;
5990#endif
5991 state->totalLen += len;
5992 XXH_ASSERT(state->bufferedSize <= XXH3_INTERNALBUFFER_SIZE);
5993
5994 /* small input : just fill in tmp buffer */
5995 if (len <= XXH3_INTERNALBUFFER_SIZE - state->bufferedSize) {
5996 XXH_memcpy(state->buffer + state->bufferedSize, input, len);
5997 state->bufferedSize += (XXH32_hash_t)len;
5998 return XXH_OK;
5999 }
6000
6001 /* total input is now > XXH3_INTERNALBUFFER_SIZE */
6002 #define XXH3_INTERNALBUFFER_STRIPES (XXH3_INTERNALBUFFER_SIZE / XXH_STRIPE_LEN)
6003 XXH_STATIC_ASSERT(XXH3_INTERNALBUFFER_SIZE % XXH_STRIPE_LEN == 0); /* clean multiple */
6004
6005 /*
6006 * Internal buffer is partially filled (always, except at beginning)
6007 * Complete it, then consume it.
6008 */
6009 if (state->bufferedSize) {
6010 size_t const loadSize = XXH3_INTERNALBUFFER_SIZE - state->bufferedSize;
6011 XXH_memcpy(state->buffer + state->bufferedSize, input, loadSize);
6012 input += loadSize;
6013 XXH3_consumeStripes(acc,
6014 &state->nbStripesSoFar, state->nbStripesPerBlock,
6015 state->buffer, XXH3_INTERNALBUFFER_STRIPES,
6016 secret, state->secretLimit,
6017 f_acc, f_scramble);
6018 state->bufferedSize = 0;
6019 }
6020 XXH_ASSERT(input < bEnd);
6021 if (bEnd - input > XXH3_INTERNALBUFFER_SIZE) {
6022 size_t nbStripes = (size_t)(bEnd - 1 - input) / XXH_STRIPE_LEN;
6023 input = XXH3_consumeStripes(acc,
6024 &state->nbStripesSoFar, state->nbStripesPerBlock,
6025 input, nbStripes,
6026 secret, state->secretLimit,
6027 f_acc, f_scramble);
6028 XXH_memcpy(state->buffer + sizeof(state->buffer) - XXH_STRIPE_LEN, input - XXH_STRIPE_LEN, XXH_STRIPE_LEN);
6029
6030 }
6031 /* Some remaining input (always) : buffer it */
6032 XXH_ASSERT(input < bEnd);
6033 XXH_ASSERT(bEnd - input <= XXH3_INTERNALBUFFER_SIZE);
6034 XXH_ASSERT(state->bufferedSize == 0);
6035 XXH_memcpy(state->buffer, input, (size_t)(bEnd-input));
6036 state->bufferedSize = (XXH32_hash_t)(bEnd-input);
6037#if defined(XXH3_STREAM_USE_STACK) && XXH3_STREAM_USE_STACK >= 1
6038 /* save stack accumulators into state */
6039 XXH_memcpy(state->acc, acc, sizeof(acc));
6040#endif
6041 }
6042
6043 return XXH_OK;
6044}

6056XXH3_digest_long (XXH64_hash_t* acc,
6057 const XXH3_state_t* state,
6058 const unsigned char* secret)

6062
6063 /*
6064 * Digest on a local copy. This way, the state remains unaltered, and it can
6065 * continue ingesting more input afterwards.
6066 */
6067 XXH_memcpy(acc, state->acc, sizeof(state->acc));

6069 /* Consume remaining stripes then point to remaining data in buffer */
6070 size_t const nbStripes = (state->bufferedSize - 1) / XXH_STRIPE_LEN;
6071 size_t nbStripesSoFar = state->nbStripesSoFar;

6073 &nbStripesSoFar, state->nbStripesPerBlock,
6074 state->buffer, nbStripes,
6075 secret, state->secretLimit,
6076 XXH3_accumulate, XXH3_scrambleAcc);
6077 lastStripePtr = state->buffer + state->bufferedSize - XXH_STRIPE_LEN;
6078 } else { /* bufferedSize < XXH_STRIPE_LEN */
6079 /* Copy to temp buffer */
6080 size_t const catchupSize = XXH_STRIPE_LEN - state->bufferedSize;
6081 XXH_ASSERT(state->bufferedSize > 0); /* there is always some input buffered */
6082 XXH_memcpy(lastStripe, state->buffer + sizeof(state->buffer) - catchupSize, catchupSize);
6083 XXH_memcpy(lastStripe + catchupSize, state->buffer, state->bufferedSize);
6084 lastStripePtr = lastStripe;
6085 }
6086 /* Last stripe */
6087 XXH3_accumulate_512(acc,
6088 lastStripePtr,
6089 secret + state->secretLimit - XXH_SECRET_LASTACC_START);
6090}

6112/* ==========================================
6113 * XXH3 128 bits (a.k.a XXH128)
6114 * ==========================================
6115 * XXH3's 128-bit variant has better mixing and strength than the 64-bit variant,
6116 * even without counting the significantly larger output size.
6117 *
6118 * For example, extra steps are taken to avoid the seed-dependent collisions
6119 * in 17-240 byte inputs (See XXH3_mix16B and XXH128_mix32B).
6120 *
6121 * This strength naturally comes at the cost of some speed, especially on short
6122 * lengths. Note that longer hashes are about as fast as the 64-bit version
6123 * due to it using only a slight modification of the 64-bit loop.
6124 *
6125 * XXH128 is also more oriented towards 64-bit machines. It is still extremely
6126 * fast for a _128-bit_ hash on 32-bit (it usually clears XXH64).
6127 */
6128
6129XXH_FORCE_INLINE XXH_PUREF XXH128_hash_t
6130XXH3_len_1to3_128b(const xxh_u8* input, size_t len, const xxh_u8* secret, XXH64_hash_t seed)
6131{
6132 /* A doubled version of 1to3_64b with different constants. */
6133 XXH_ASSERT(input != NULL);
6134 XXH_ASSERT(1 <= len && len <= 3);
6135 XXH_ASSERT(secret != NULL);
6136 /*
6137 * len = 1: combinedl = { input[0], 0x01, input[0], input[0] }
6138 * len = 2: combinedl = { input[1], 0x02, input[0], input[1] }
6139 * len = 3: combinedl = { input[2], 0x03, input[0], input[1] }
6140 */
6141 { xxh_u8 const c1 = input[0];
6142 xxh_u8 const c2 = input[len >> 1];
6143 xxh_u8 const c3 = input[len - 1];
6144 xxh_u32 const combinedl = ((xxh_u32)c1 <<16) | ((xxh_u32)c2 << 24)
6145 | ((xxh_u32)c3 << 0) | ((xxh_u32)len << 8);
6146 xxh_u32 const combinedh = XXH_rotl32(XXH_swap32(combinedl), 13);
6147 xxh_u64 const bitflipl = (XXH_readLE32(secret) ^ XXH_readLE32(secret+4)) + seed;
6148 xxh_u64 const bitfliph = (XXH_readLE32(secret+8) ^ XXH_readLE32(secret+12)) - seed;
6149 xxh_u64 const keyed_lo = (xxh_u64)combinedl ^ bitflipl;
6150 xxh_u64 const keyed_hi = (xxh_u64)combinedh ^ bitfliph;
6151 XXH128_hash_t h128;
6152 h128.low64 = XXH64_avalanche(keyed_lo);
6153 h128.high64 = XXH64_avalanche(keyed_hi);
6154 return h128;
6155 }
6156}

6158XXH_FORCE_INLINE XXH_PUREF XXH128_hash_t
6159XXH3_len_4to8_128b(const xxh_u8* input, size_t len, const xxh_u8* secret, XXH64_hash_t seed)
6160{
6161 XXH_ASSERT(input != NULL);
6162 XXH_ASSERT(secret != NULL);
6163 XXH_ASSERT(4 <= len && len <= 8);
6164 seed ^= (xxh_u64)XXH_swap32((xxh_u32)seed) << 32;
6165 { xxh_u32 const input_lo = XXH_readLE32(input);
6166 xxh_u32 const input_hi = XXH_readLE32(input + len - 4);
6167 xxh_u64 const input_64 = input_lo + ((xxh_u64)input_hi << 32);
6168 xxh_u64 const bitflip = (XXH_readLE64(secret+16) ^ XXH_readLE64(secret+24)) + seed;
6169 xxh_u64 const keyed = input_64 ^ bitflip;
6170
6171 /* Shift len to the left to ensure it is even, this avoids even multiplies. */
6172 XXH128_hash_t m128 = XXH_mult64to128(keyed, XXH_PRIME64_1 + (len << 2));
6173
6174 m128.high64 += (m128.low64 << 1);
6175 m128.low64 ^= (m128.high64 >> 3);
6176
6177 m128.low64 = XXH_xorshift64(m128.low64, 35);
6178 m128.low64 *= PRIME_MX2;
6179 m128.low64 = XXH_xorshift64(m128.low64, 28);
6180 m128.high64 = XXH3_avalanche(m128.high64);
6181 return m128;
6182 }
6183}

6185XXH_FORCE_INLINE XXH_PUREF XXH128_hash_t
6186XXH3_len_9to16_128b(const xxh_u8* input, size_t len, const xxh_u8* secret, XXH64_hash_t seed)
6187{
6188 XXH_ASSERT(input != NULL);
6189 XXH_ASSERT(secret != NULL);
6190 XXH_ASSERT(9 <= len && len <= 16);
6191 { xxh_u64 const bitflipl = (XXH_readLE64(secret+32) ^ XXH_readLE64(secret+40)) - seed;
6192 xxh_u64 const bitfliph = (XXH_readLE64(secret+48) ^ XXH_readLE64(secret+56)) + seed;
6193 xxh_u64 const input_lo = XXH_readLE64(input);
6194 xxh_u64 input_hi = XXH_readLE64(input + len - 8);
6195 XXH128_hash_t m128 = XXH_mult64to128(input_lo ^ input_hi ^ bitflipl, XXH_PRIME64_1);
6196 /*
6197 * Put len in the middle of m128 to ensure that the length gets mixed to
6198 * both the low and high bits in the 128x64 multiply below.
6199 */
6200 m128.low64 += (xxh_u64)(len - 1) << 54;
6201 input_hi ^= bitfliph;
6202 /*
6203 * Add the high 32 bits of input_hi to the high 32 bits of m128, then
6204 * add the long product of the low 32 bits of input_hi and XXH_PRIME32_2 to
6205 * the high 64 bits of m128.
6206 *
6207 * The best approach to this operation is different on 32-bit and 64-bit.
6208 */
6209 if (sizeof(void *) < sizeof(xxh_u64)) { /* 32-bit */
6210 /*
6211 * 32-bit optimized version, which is more readable.
6212 *
6213 * On 32-bit, it removes an ADC and delays a dependency between the two
6214 * halves of m128.high64, but it generates an extra mask on 64-bit.
6215 */
6216 m128.high64 += (input_hi & 0xFFFFFFFF00000000ULL) + XXH_mult32to64((xxh_u32)input_hi, XXH_PRIME32_2);
6217 } else {
6218 /*
6219 * 64-bit optimized (albeit more confusing) version.
6220 *
6221 * Uses some properties of addition and multiplication to remove the mask:
6222 *
6223 * Let:
6224 * a = input_hi.lo = (input_hi & 0x00000000FFFFFFFF)
6225 * b = input_hi.hi = (input_hi & 0xFFFFFFFF00000000)
6226 * c = XXH_PRIME32_2
6227 *
6228 * a + (b * c)
6229 * Inverse Property: x + y - x == y
6230 * a + (b * (1 + c - 1))
6231 * Distributive Property: x * (y + z) == (x * y) + (x * z)
6232 * a + (b * 1) + (b * (c - 1))
6233 * Identity Property: x * 1 == x
6234 * a + b + (b * (c - 1))
6235 *
6236 * Substitute a, b, and c:
6237 * input_hi.hi + input_hi.lo + ((xxh_u64)input_hi.lo * (XXH_PRIME32_2 - 1))
6238 *
6239 * Since input_hi.hi + input_hi.lo == input_hi, we get this:
6240 * input_hi + ((xxh_u64)input_hi.lo * (XXH_PRIME32_2 - 1))
6241 */
6242 m128.high64 += input_hi + XXH_mult32to64((xxh_u32)input_hi, XXH_PRIME32_2 - 1);
6243 }
6244 /* m128 ^= XXH_swap64(m128 >> 64); */
6245 m128.low64 ^= XXH_swap64(m128.high64);
6246
6247 { /* 128x64 multiply: h128 = m128 * XXH_PRIME64_2; */
6248 XXH128_hash_t h128 = XXH_mult64to128(m128.low64, XXH_PRIME64_2);
6249 h128.high64 += m128.high64 * XXH_PRIME64_2;
6250
6251 h128.low64 = XXH3_avalanche(h128.low64);
